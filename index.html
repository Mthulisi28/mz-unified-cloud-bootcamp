# ============================================================ # MZ UNIFIED CLOUD ACADEMY — AUTOMATED DEPLOYMENT PIPELINE # Version : 2.0.0 # Author : Mthulisi Zulu | green28@live.co.za # Domain : quickreserve.co.za / mthulisi28.github.io # Run as : PowerShell 5.1+ (right-click > Run with PowerShell) # ============================================================ Set-StrictMode -Version Latest $ErrorActionPreference = "Stop" # ── CONFIGURATION ──────────────────────────────────────────── $CONFIG = @{ # Paths SourceRoot = "C:\Users\green\OneDrive\Desktop\bootcamp" RepoRoot = "C:\bootcamp\mz-unified-cloud-bootcamp" AssetsDir = "C:\bootcamp\mz-unified-cloud-bootcamp\assets" # Git RemoteURL = "https://github.com/Mthulisi28/mz-unified-cloud-bootcamp.git" Branch = "main" # Identity GitEmail = "green28@live.co.za" GitName = "Mthulisi Zulu" # Portal LiveURL = "https://mthulisi28.github.io/mz-unified-cloud-bootcamp/" Version = "3.2.0" } # ── COLOUR PALETTE ─────────────────────────────────────────── function Write-Gold { param($msg) Write-Host $msg -ForegroundColor Yellow } function Write-OK { param($msg) Write-Host " [OK] $msg" -ForegroundColor Green } function Write-Fail { param($msg) Write-Host " [!!] $msg" -ForegroundColor Red } function Write-Info { param($msg) Write-Host " [..] $msg" -ForegroundColor Cyan } function Write-Warn { param($msg) Write-Host " [??] $msg" -ForegroundColor Magenta } function Show-Banner { Write-Host "" Write-Gold " ╔══════════════════════════════════════════════════════╗" Write-Gold " ║ MZ UNIFIED CLOUD ACADEMY — DEPLOY PIPELINE v2 ║" Write-Gold " ║ Mthulisi Zulu · Cloud Governance Architect ║" Write-Gold " ╚══════════════════════════════════════════════════════╝" Write-Host "" } # ── STEP 0 — PRE-FLIGHT CHECK ──────────────────────────────── function Step-PreFlight { Write-Gold "► STEP 0 — PRE-FLIGHT CHECK" # Git present? try { $v = git --version 2>&1 Write-OK "Git found: $v" } catch { Write-Fail "Git not found. Install from https://git-scm.com" exit 1 } # Repo exists? if (-not (Test-Path $CONFIG.RepoRoot)) { Write-Warn "Repo not found locally. Cloning now..." git clone $CONFIG.RemoteURL $CONFIG.RepoRoot Write-OK "Repo cloned to $($CONFIG.RepoRoot)" } else { Write-OK "Repo found at $($CONFIG.RepoRoot)" } # Assets dir if (-not (Test-Path $CONFIG.AssetsDir)) { New-Item -ItemType Directory -Path $CONFIG.AssetsDir | Out-Null Write-OK "Created assets directory" } else { Write-OK "Assets directory ready" } # Source exists? if (-not (Test-Path $CONFIG.SourceRoot)) { Write-Fail "Source folder not found: $($CONFIG.SourceRoot)" Write-Warn "Update SourceRoot in CONFIG at top of script." exit 1 } else { Write-OK "Source folder found: $($CONFIG.SourceRoot)" } # Git identity Set-Location $CONFIG.RepoRoot git config user.email $CONFIG.GitEmail | Out-Null git config user.name $CONFIG.GitName | Out-Null Write-OK "Git identity set: $($CONFIG.GitName) <$($CONFIG.GitEmail)>" Write-Host "" } # ── STEP 1 — SCAN SOURCE FOR NEW ASSETS ───────────────────── function Step-ScanAssets { Write-Gold "► STEP 1 — SCANNING SOURCE FOR NEW / UPDATED ASSETS" $sourceFiles = Get-ChildItem -Path $CONFIG.SourceRoot -File -Recurse | Where-Object { $_.Extension -match "\.(pdf|docx|xlsx|pptx|png|jpg|mp4)$" } $stagedFiles = @() $skippedFiles = @() foreach ($file in $sourceFiles) { $dest = Join-Path $CONFIG.AssetsDir $file.Name if (-not (Test-Path $dest)) { # New file — always copy $stagedFiles += $file } else { # Existing — copy only if source is newer $destItem = Get-Item $dest if ($file.LastWriteTime -gt $destItem.LastWriteTime) { $stagedFiles += $file } else { $skippedFiles += $file } } } Write-Info "Source files found : $($sourceFiles.Count)" Write-Info "New / updated files : $($stagedFiles.Count)" Write-Info "Already up-to-date : $($skippedFiles.Count)" Write-Host "" if ($stagedFiles.Count -eq 0) { Write-Warn "No new files to deploy. All assets are current." Write-Host "" return $null } # Show what will be copied Write-Gold " FILES TO DEPLOY:" foreach ($f in $stagedFiles) { $kb = [math]::Round($f.Length / 1KB, 1) Write-Host " + $($f.Name) ($kb KB)" -ForegroundColor White } Write-Host "" return $stagedFiles } # ── STEP 2 — COPY AND VERIFY ASSETS ───────────────────────── function Step-CopyAssets($files) { Write-Gold "► STEP 2 — COPYING AND VERIFYING ASSETS" $verified = @() $failed = @() foreach ($file in $files) { $dest = Join-Path $CONFIG.AssetsDir $file.Name try { Copy-Item $file.FullName -Destination $dest -Force # Verify: compare file sizes $destItem = Get-Item $dest if ($destItem.Length -eq $file.Length) { Write-OK "$($file.Name)" $verified += $file.Name } else { Write-Fail "Size mismatch: $($file.Name) — source $($file.Length)b vs dest $($destItem.Length)b" $failed += $file.Name } } catch { Write-Fail "Copy failed: $($file.Name) — $_" $failed += $file.Name } } Write-Host "" Write-Info "Verified OK : $($verified.Count)" if ($failed.Count -gt 0) { Write-Fail "Failed : $($failed.Count)" foreach ($f in $failed) { Write-Fail " - $f" } Write-Warn "Resolve failed files before pushing. Aborting." exit 1 } Write-Host "" return $verified } # ── STEP 3 — GIT STAGE, COMMIT, PUSH ──────────────────────── function Step-GitPush($fileNames, $label) { Write-Gold "► STEP 3 — GIT COMMIT AND PUSH" Set-Location $CONFIG.RepoRoot # Pull latest first (avoid push rejection) Write-Info "Pulling latest from origin/$($CONFIG.Branch)..." git pull origin $CONFIG.Branch --quiet 2>&1 | Out-Null Write-OK "Pull complete" # Stage assets git add assets/ 2>&1 | Out-Null # Stage index.html if it was modified $indexPath = Join-Path $CONFIG.RepoRoot "index.html" if (Test-Path $indexPath) { $status = git status --short index.html 2>&1 if ($status -match "M|A") { git add index.html 2>&1 | Out-Null Write-OK "index.html staged" } } # Build commit message $ts = Get-Date -Format "yyyy-MM-dd HH:mm" $vTag = "uca-v$($CONFIG.Version)" $summary = if ($label) { $label } else { "$($fileNames.Count) asset(s) deployed" } $msg = "feat(assets): $summary — $vTag — $ts" # Check if there's anything to commit $staged = git status --short 2>&1 if (-not $staged) { Write-Warn "Nothing staged — working tree is clean. No commit needed." Write-Host "" return } git commit -m $msg 2>&1 | ForEach-Object { Write-Host " $_" -ForegroundColor DarkGray } Write-OK "Committed: $msg" Write-Info "Pushing to $($CONFIG.RemoteURL) / $($CONFIG.Branch)..." git push origin $CONFIG.Branch 2>&1 | ForEach-Object { Write-Host " $_" -ForegroundColor DarkGray } Write-OK "Push complete" Write-Host "" } # ── STEP 4 — DEPLOY SUMMARY ───────────────────────────────── function Step-Summary($fileNames) { Write-Gold "► STEP 4 — DEPLOYMENT SUMMARY" Write-OK "Portal live at : $($CONFIG.LiveURL)" Write-OK "Files deployed : $($fileNames.Count)" Write-OK "Timestamp : $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss')" Write-Host "" Write-Gold " ════════════════════════════════════════════════" Write-Gold " MACHINE UPDATED. COHORT CAN ACCESS MATERIALS." Write-Gold " ════════════════════════════════════════════════" Write-Host "" # Open portal in browser $open = Read-Host " Open live portal in browser? (Y/N)" if ($open -eq "Y" -or $open -eq "y") { Start-Process $CONFIG.LiveURL } } # ── SINGLE-FILE MODE ───────────────────────────────────────── # Drop ONE specific file by name — useful for targeted sessions function Deploy-SingleFile($fileName, $commitLabel) { $src = Join-Path $CONFIG.SourceRoot $fileName if (-not (Test-Path $src)) { Write-Fail "File not found in source: $fileName" Write-Warn "Place it in: $($CONFIG.SourceRoot)" exit 1 } $fakeList = @([System.IO.FileInfo](Get-Item $src)) $verified = Step-CopyAssets $fakeList Step-GitPush $verified $commitLabel Step-Summary $verified } # ── MAIN ───────────────────────────────────────────────────── Show-Banner Step-PreFlight Write-Host " SELECT DEPLOYMENT MODE:" -ForegroundColor White Write-Host " [1] Full Sync — stage ALL new/updated assets from OneDrive" -ForegroundColor Cyan Write-Host " [2] Single File — deploy one named file (e.g. SESSION 6)" -ForegroundColor Cyan Write-Host " [3] Portal Only — push index.html (no asset changes)" -ForegroundColor Cyan Write-Host " [Q] Quit" -ForegroundColor DarkGray Write-Host "" $mode = Read-Host " Choice" switch ($mode.ToUpper()) { "1" { $files = Step-ScanAssets if ($null -eq $files) { exit 0 } $verified = Step-CopyAssets $files Step-GitPush $verified $null Step-Summary $verified } "2" { Write-Host "" $fname = Read-Host " Enter exact filename (e.g. SESSION 6 — DATABASES.pdf)" $label = Read-Host " Commit label (e.g. deploy session 6 databases)" Deploy-SingleFile $fname $label } "3" { Write-Gold "► PORTAL-ONLY PUSH" Set-Location $CONFIG.RepoRoot $indexPath = Join-Path $CONFIG.RepoRoot "index.html" if (-not (Test-Path $indexPath)) { Write-Fail "index.html not found in repo root." exit 1 } git add index.html 2>&1 | Out-Null $staged = git status --short 2>&1 if (-not $staged) { Write-Warn "index.html unchanged — nothing to push." exit 0 } $ts = Get-Date -Format "yyyy-MM-dd HH:mm" $msg = "feat(portal): portal update — uca-v$($CONFIG.Version) — $ts" git commit -m $msg 2>&1 | ForEach-Object { Write-Host " $_" -ForegroundColor DarkGray } git push origin $CONFIG.Branch 2>&1 | ForEach-Object { Write-Host " $_" -ForegroundColor DarkGray } Write-OK "Portal pushed: $($CONFIG.LiveURL)" Write-Host "" } "Q" { Write-Host " Exited. No changes made." -ForegroundColor DarkGray; exit 0 } default { Write-Fail "Invalid choice. Run script again." ; exit 1 } }
